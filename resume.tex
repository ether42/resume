\unexpanded\def\Job#1#2#3#4#5{ % macro for job formatting
  \startJobTitle
  \textrule[top]{#1} % job name
  \stopJobTitle
  \startJobInfos
  #2 % job location
  \JobInfos
  #3 % job date
  \stopJobInfos
  #4 % job description
  \startitemize[2,packed,joinedup]
  #5 % job activities
  \stopitemize
}

  \head #1 % following blank line is needed

  \stopitemize
}
