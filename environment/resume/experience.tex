\startenvironment[experience]

  \defineparagraphs[exptitle][
    n=1,
    before=\blank,
    after=\nowhitespace,
  ]
  \setupparagraphs[exptitle][1][
    align=flushleft,
    style=bold,
    width=\hsize,
    distance=0cm,
  ]
  \define[1]\experiencetitle{
    \startexptitle
      \textrule[top]{#1}
    \stopexptitle
  }

  \defineparagraphs[explocality][
    n=2,
    before=\nowhitespace,
    after=\nowhitespace,
  ]
  \setupparagraphs[explocality][1][
    align=flushleft,
    style=italic,
    distance=0cm,
  ]
  \setupparagraphs[explocality][2][
    align=flushright,
    style=italic,
    distance=0cm,
  ]
  \define[2]\experiencelocality{
    \startexplocality
    #1
    \explocality
    #2
    \stopexplocality
  }

  \defineitemgroup[experienceactivity]
  \setupitemgroup[experienceactivity][each][
    2, % hyphen symbol
    packed, % no space between items
    joinedup, % no space before and after
  ]

  \define[1]\experiencecategory{\inleft{#1}}

\stopenvironment
